\begin{frame}{data challenge revisited}{insufficiency vs. representativeness}
		\hspace{-4mm}moderate improvements can be made to deal with insufficient data, but
		\visible<2->{
		\bigskip
		\question{is the amount of data really the main issue}}
		
		\begin{itemize}
				\item	a closer look at music datasets for popular tasks
				\begin{itemize}
				\item	chord detection: Beatles dataset
					\begin{itemize}
						\item audio data
						\item	chord vocabulary
					\end{itemize}
				\item	genre classification: several
					\begin{itemize}
						\item genre labeling
					\end{itemize}
				\item	source separation: MUSDB
					\begin{itemize}
						\item choice of classes
					\end{itemize}
				\end{itemize}
				\item homogeneity questions representativeness of data
				\item	category restrictions question generalization to task
				\item problem for methodology: data might impact architectural choices
				\item	problem for training: system cannot learn what is hasn't seen or what seems irrelevant
				\item	problem for testing: results may imply good performance yet cannot be generalized
		\end{itemize}
\end{frame}
